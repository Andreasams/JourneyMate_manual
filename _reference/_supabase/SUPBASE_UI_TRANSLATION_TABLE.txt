create table public.ui_translations (
  id bigint generated always as identity not null,
  language_code text not null,
  translation_key text not null,
  translation_value text not null,
  category text not null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint flutterflowtranslations_pkey primary key (id),
  constraint flutterflowtranslations_lang_key_unique unique (language_code, translation_key),
  constraint translations_unique_lang_key unique (language_code, translation_key),
  constraint check_valid_language_code check (
    (
      language_code = any (
        array[
          'en'::text,
          'da'::text,
          'de'::text,
          'es'::text,
          'fi'::text,
          'fr'::text,
          'it'::text,
          'ja'::text,
          'ko'::text,
          'nl'::text,
          'no'::text,
          'pl'::text,
          'sv'::text,
          'uk'::text,
          'zh'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_translations_category on public.ui_translations using btree (category) TABLESPACE pg_default;

create index IF not exists idx_translations_lang_key on public.ui_translations using btree (language_code, translation_key) TABLESPACE pg_default;
