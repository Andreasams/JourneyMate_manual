create table public."BusinessInfo" (
  business_id integer generated by default as identity not null,
  business_name text not null,
  is_active boolean not null default false,
  typesense_id integer null,
  brand_id integer null,
  company_id integer null,
  business_type_id integer null,
  business_type text null,
  last_reviewed_at timestamp with time zone null,
  price_range_min numeric null,
  price_range_max numeric null,
  price_range_currency_code character(3) null default 'DKK'::bpchar,
  next_menu_review_date date null,
  menu_review_notes text null,
  created_at timestamp with time zone not null default now(),
  constraint businesses_pkey primary key (business_id),
  constraint business_info_typesense_id_key unique (typesense_id),
  constraint BusinessInfo_brand_id_fkey foreign KEY (brand_id) references "Brand" (brand_id) on delete CASCADE,
  constraint BusinessInfo_company_id_fkey foreign KEY (company_id) references "Company" (company_id) on delete set null,
  constraint fk_businessinfo_filter_id foreign KEY (business_type_id) references "Filter" (filter_id) on update CASCADE on delete RESTRICT,
  constraint chk_price_range_logical check ((price_range_max >= price_range_min)),
  constraint chk_price_range_max_positive check ((price_range_max >= (0)::numeric)),
  constraint chk_price_range_min_positive check ((price_range_min >= (0)::numeric)),
  constraint check_valid_brand_id check (
    (
      (brand_id is null)
      or (brand_id > 0)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_business_info_brand_id_fk on public."BusinessInfo" using btree (brand_id) TABLESPACE pg_default;

create index IF not exists idx_business_info_business_type_fk on public."BusinessInfo" using btree (business_type_id) TABLESPACE pg_default;

create index IF not exists idx_business_info_company_id_fk on public."BusinessInfo" using btree (company_id) TABLESPACE pg_default;

create index IF not exists idx_businessinfo_currency on public."BusinessInfo" using btree (price_range_currency_code) TABLESPACE pg_default;

create index IF not exists idx_businessinfo_price_range_max on public."BusinessInfo" using btree (price_range_max) TABLESPACE pg_default;

create index IF not exists idx_businessinfo_price_range_min on public."BusinessInfo" using btree (price_range_min) TABLESPACE pg_default;
