
/*
FUNCTION PURPOSE: Retrieves complete business information with language-specific translations
ROLE IN SYSTEM: Primary API endpoint for client applications to fetch all business data including
  profile, gallery, menu structure, exchange rates, filters, and business hours
BUSINESS LOGIC: Aggregates business data from multiple tables and returns language-specific
  translations for business_type, tags, descriptions, menu items

STEPS:
1. Retrieve business profile with language-specific business_type from BusinessInfo_LanguageTags
2. Get all language-specific tags from BusinessInfo_LanguageTags for the requested language
3. Fetch business description with language fallback (translation table -> base table -> empty string)
4. Include address, contact information, brand/company data, and profile picture
5. Aggregate gallery images with category information
6. Build menu structure with translated menu and category names/descriptions
7. Retrieve exchange rate for the language's associated currency
8. Aggregate raw filters from business_x_filter (client widgets handle their own filtering/display)
9. Retrieve business_hours + open_windows via get_business_hours_for_typesense()
10. Combine all data into single JSONB response with success status

LANGUAGE-SPECIFIC FIELDS:
- business_type: Translated from BusinessInfo_LanguageTags.business_type_text
- tags: From BusinessInfo_LanguageTags.tags (language-specific filter tags)
- description: From BusinessDescriptionTranslation (falls back to BusinessDescription)
- menu titles/descriptions: From MenuTranslation and MenuCategoryTranslation

CHANGES vs previous version:
- Added 'filters' array to business_profile: raw rows from business_x_filter
  (filter_id, filter_name, filter_category_id). Each client widget applies its
  own display/exclusion logic — no server-side filtering applied.
- Added 'business_hours' to top-level response: full hours object keyed by
  day_of_week (0-6), delegated to get_business_hours_for_typesense() to keep
  overnight-splitting logic in one place.
- Added 'open_windows' to top-level response: pre-computed [{day, open, close}]
  array in minutes-since-midnight, also from get_business_hours_for_typesense().

DEPENDENCIES: BusinessInfo, BusinessInfo_LanguageTags, BusinessDescription,
  BusinessDescriptionTranslation, BusinessAddress, GeographyCity,
  BusinessProfilePicture, Brand, Company, business_x_url,
  business_x_phone_number, business_x_email, GalleryImage, Menu,
  MenuTranslation, MenuCategory, MenuCategoryTranslation, ExchangeRate,
  business_x_filter, get_business_hours_for_typesense()
TRIGGERS: None - read-only query function
USAGE: SELECT get_business_complete_info(4130, 'da') for Danish translations
*/
DECLARE
    v_business_profile JSONB;
    v_gallery          JSONB;
    v_menu_structure   JSONB;
    v_exchange_rate    JSONB;
    v_filters          JSONB;
    v_hours_result     JSONB;
    v_result           JSONB;
BEGIN
    RAISE LOG 'get_business_complete_info: Starting for business_id % with language %', p_business_id, p_language_code;

    -- =========================================================================
    -- BUSINESS PROFILE
    -- =========================================================================
    SELECT jsonb_build_object(
        'business_id',             bi.business_id,
        'business_name',           bi.business_name,
        'business_type', COALESCE(
            (SELECT business_type_text
             FROM "BusinessInfo_LanguageTags"
             WHERE business_id = bi.business_id
             AND language_code = p_language_code),
            bi.business_type
        ),
        'price_range_min',         bi.price_range_min,
        'price_range_max',         bi.price_range_max,
        'price_range_currency_code', bi.price_range_currency_code,
        'is_active',               bi.is_active,

        'tags', COALESCE(
            (SELECT tags
             FROM "BusinessInfo_LanguageTags"
             WHERE business_id = bi.business_id
             AND language_code = p_language_code),
            '[]'::jsonb
        ),

        'typesense_id',            bi.typesense_id,
        'last_reviewed_at',        bi.last_reviewed_at,

        'description', COALESCE(
            (SELECT bdt.description
             FROM "BusinessDescription" bd
             JOIN "BusinessDescriptionTranslation" bdt ON bd.translation_id = bdt.translation_id
             WHERE bd.business_id = bi.business_id
             AND bdt.language_code = p_language_code),
            (SELECT bd.description
             FROM "BusinessDescription" bd
             WHERE bd.business_id = bi.business_id
             AND bd.language_code = p_language_code),
            ''
        ),

        'address', jsonb_build_object(
            'address_id',          ba.address_id,
            'street',              ba.street,
            'postal_code',         ba.postal_code,
            'city_id',             ba.city_id,
            'city_name',           gc.name,
            'postal_city',         ba.postal_city_cph_frederiksberg,
            'neighbourhood_name',  ba.neighbourhood_name,
            'latitude',            ba.latitude,
            'longitude',           ba.longitude,
            'location',            ba.location,
            'google_maps_url',     ba.google_maps_url
        ),

        'profile_picture', jsonb_build_object(
            'business_profile_picture_id', bpp.business_profile_picture_id,
            'url',                 bpp.url,
            'name',                bpp.name,
            'alt_text',            bpp.alt_text,
            'created_at',          bpp.created_at,
            'updated_at',          bpp.updated_at
        ),

        'contact', jsonb_build_object(
            'website_url',         website_url.url,
            'facebook_url',        facebook_url.url,
            'instagram_url',       instagram_url.url,
            'reservation_url',     reservation_url.url,
            'general_phone',       general_phone.phone_number,
            'reservation_phone',   reservation_phone.phone_number,
            'general_email',       general_email.email,
            'reservation_email',   reservation_email.email
        ),

        'brand', jsonb_build_object(
            'brand_id',            bi.brand_id,
            'brand_name',          b.name
        ),
        'company', jsonb_build_object(
            'company_id',          bi.company_id,
            'company_name',        c.name
        )
    ) INTO v_business_profile
    FROM "BusinessInfo" bi
    LEFT JOIN "BusinessAddress"        ba  ON bi.business_id = ba.business_id
    LEFT JOIN "GeographyCity"          gc  ON ba.city_id = gc.city_id
    LEFT JOIN "BusinessProfilePicture" bpp ON bi.business_id = bpp.business_id
    LEFT JOIN "Brand"                  b   ON bi.brand_id = b.brand_id
    LEFT JOIN "Company"                c   ON bi.company_id = c.company_id
    LEFT JOIN business_x_url           website_url     ON bi.business_id = website_url.business_id     AND website_url.url_type_id = 1
    LEFT JOIN business_x_url           facebook_url    ON bi.business_id = facebook_url.business_id    AND facebook_url.url_type_id = 2
    LEFT JOIN business_x_url           instagram_url   ON bi.business_id = instagram_url.business_id   AND instagram_url.url_type_id = 3
    LEFT JOIN business_x_url           reservation_url ON bi.business_id = reservation_url.business_id AND reservation_url.url_type_id = 4
    LEFT JOIN business_x_phone_number  general_phone     ON bi.business_id = general_phone.business_id     AND general_phone.phone_type_id = 1
    LEFT JOIN business_x_phone_number  reservation_phone ON bi.business_id = reservation_phone.business_id AND reservation_phone.phone_type_id = 2
    LEFT JOIN business_x_email         general_email     ON bi.business_id = general_email.business_id     AND general_email.email_type_id = 1
    LEFT JOIN business_x_email         reservation_email ON bi.business_id = reservation_email.business_id AND reservation_email.email_type_id = 2
    WHERE bi.business_id = p_business_id;

    -- =========================================================================
    -- FILTERS
    -- Raw rows from business_x_filter — no server-side exclusions applied.
    -- Each client widget (BusinessFeatureButtons, payment widget, etc.)
    -- applies its own display and exclusion logic independently.
    -- =========================================================================
    SELECT COALESCE(
        jsonb_agg(
            jsonb_build_object(
                'filter_id',          bxf.filter_id,
                'filter_name',        bxf.filter_name,
                'filter_category_id', bxf.filter_category_id
            )
            ORDER BY bxf.filter_category_id, bxf.filter_id
        ),
        '[]'::jsonb
    ) INTO v_filters
    FROM "business_x_filter" bxf
    WHERE bxf.business_id = p_business_id;

    -- =========================================================================
    -- GALLERY
    -- =========================================================================
    SELECT jsonb_agg(
        jsonb_build_object(
            'image_id',     gi.image_id,
            'image_url',    gi.image_url,
            'image_name',   gi.image_name,
            'alt_text',     gi.alt_text,
            'description',  gi.description,
            'created_at',   gi.created_at,
            'updated_at',   gi.updated_at,
            'category_id',  gi.category_id,
            'category_name', CASE
                WHEN gi.category_id = 1 THEN 'Interior'
                WHEN gi.category_id = 2 THEN 'Outdoor'
                WHEN gi.category_id = 3 THEN 'Food'
                WHEN gi.category_id = 4 THEN 'Menu'
                ELSE NULL
            END
        )
    ) INTO v_gallery
    FROM "GalleryImage" gi
    WHERE gi.business_id = p_business_id;

    -- =========================================================================
    -- MENU STRUCTURE
    -- =========================================================================
    SELECT jsonb_agg(
        jsonb_build_object(
            'menu_id',           m.menu_id,
            'menu_title', COALESCE(
                (SELECT mt.name
                 FROM "MenuTranslation" mt
                 WHERE mt.translation_id = m.translation_id
                 AND mt.language_code = p_language_code),
                m.name
            ),
            'menu_description', COALESCE(
                (SELECT mt.description
                 FROM "MenuTranslation" mt
                 WHERE mt.translation_id = m.translation_id
                 AND mt.language_code = p_language_code),
                m.description
            ),
            'menu_display_order',    m.display_order,
            'business_id',           m.business_id,
            'menu_category_id',      mc.menu_category_id,
            'category_name', COALESCE(
                (SELECT mct.name
                 FROM "MenuCategoryTranslation" mct
                 WHERE mct.translation_id = mc.translation_id
                 AND mct.language_code = p_language_code),
                mc.name
            ),
            'category_description', COALESCE(
                (SELECT mct.description
                 FROM "MenuCategoryTranslation" mct
                 WHERE mct.translation_id = mc.translation_id
                 AND mct.language_code = p_language_code),
                mc.description
            ),
            'category_display_order', mc.display_order,
            'category_type',          mc.purpose,
            'is_beverage',            mc.is_beverage
        )
        ORDER BY m.display_order, mc.display_order
    ) INTO v_menu_structure
    FROM "Menu" m
    JOIN "MenuCategory" mc ON m.menu_id = mc.menu_id
    WHERE m.business_id = p_business_id
    AND mc.purpose IN ('menu_category', 'menu_package');

    -- =========================================================================
    -- EXCHANGE RATE
    -- =========================================================================
    SELECT jsonb_build_object(
        'from_currency', er.from_currency,
        'to_currency',   er.to_currency,
        'rate',          er.rate,
        'last_updated',  er.last_updated,
        'language_code', p_language_code
    ) INTO v_exchange_rate
    FROM "ExchangeRate" er
    WHERE p_language_code = ANY(er.associated_language_code)
    LIMIT 1;

    -- =========================================================================
    -- BUSINESS HOURS + OPEN_WINDOWS
    -- Delegated to get_business_hours_for_typesense() — single source of truth
    -- for overnight-splitting logic and hours aggregation.
    -- =========================================================================
    v_hours_result := get_business_hours_for_typesense(p_business_id)::jsonb;

    -- =========================================================================
    -- COMBINE
    -- =========================================================================
    v_result := jsonb_build_object(
        'success',          true,
        'business_profile', COALESCE(v_business_profile, '{}'::jsonb) || jsonb_build_object('filters', v_filters),
        'gallery',          COALESCE(v_gallery, '[]'::jsonb),
        'menu_structure',   COALESCE(v_menu_structure, '[]'::jsonb),
        'exchange_rate',    COALESCE(v_exchange_rate, 'null'::jsonb),
        'business_hours',   COALESCE(v_hours_result->'business_hours', '{}'::jsonb),
        'open_windows',     COALESCE(v_hours_result->'open_windows', '[]'::jsonb),
        'language_code',    p_language_code,
        'retrieved_at',     NOW()
    );

    RAISE LOG 'get_business_complete_info: Successfully completed for business_id % language %', p_business_id, p_language_code;

    RETURN v_result;

EXCEPTION WHEN OTHERS THEN
    RAISE LOG 'get_business_complete_info: ERROR for business_id %: % (SQLSTATE: %)',
        p_business_id, SQLERRM, SQLSTATE;

    RETURN jsonb_build_object(
        'success',      false,
        'error',        SQLERRM,
        'business_id',  p_business_id,
        'language_code', p_language_code
    );
END;
