INPUTS:
* String: message
* String: contact
* String: languageCode
* String: topic
* Boolean: allowContact
* String: name

create table public."zUserFormShareFeedback" (
  id bigint generated by default as identity not null,
  received_at timestamp with time zone not null default now(),
  topic text not null,
  page text not null,
  message text not null,
  user_name text null,
  contact_details text null,
  language_code text null,
  allow_contact boolean null,
  constraint zUserFormShareFeedback_pkey primary key (id)
) TABLESPACE pg_default;

create index IF not exists idx_user_feedback_received_at on public."zUserFormShareFeedback" using btree (received_at desc) TABLESPACE pg_default;
