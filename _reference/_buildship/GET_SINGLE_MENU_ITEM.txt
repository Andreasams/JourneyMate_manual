import { createClient } from '@supabase/supabase-js';

interface NodeInputs {
  supabaseUrl: string;
  supabaseKey: string;
  menuItemId: number;
  languageCode: string;
}

interface NodeScriptOptions {
  logging: any;
  env: any;
  getBuildShipFile: any;
}

interface MenuItemOutput {
  menu_item_id?: number;
  item_name?: string;
  item_description?: string;
  base_price?: number;
  is_price_per_person?: boolean;
  item_image_url?: string;
  is_beverage?: boolean;
  display_order?: number;
  language_code?: string;
  business_id?: number;
  authentic_languages?: string[];
  dietary_type_ids?: number[];
  dietary_type_can_be_made_ids?: number[];
  allergy_ids?: number[];
  item_modifier_groups?: any[];
  error?: string;
}

/**
 * Retrieves a single menu item in a specified language.
 * 
 * Returns the same structure as a single item from get_menu_complete, enabling
 * seamless integration with existing menu item display widgets.
 * 
 * Use Case: "View in original language" feature
 * When a user views a menu item translated to their preferred language (e.g., English)
 * but wants to see the original version (e.g., Danish), this endpoint retrieves
 * the item with all translations in the requested authentic language.
 * 
 * The menu item's authentic_languages array indicates which languages are available
 * for this feature (typically 1-3 language codes from the parent Menu).
 * 
 * Parameters:
 * - menuItemId: The ID of the menu item to retrieve
 * - languageCode: The language code to retrieve translations in (typically one of the authentic_languages)
 * 
 * Returns: Complete menu item object with translated names/descriptions for the item,
 * modifier groups, and modifiers. All other fields (prices, IDs, arrays) are unchanged.
 */
export default async function getMenuItemInLanguage(
  { supabaseUrl, supabaseKey, menuItemId, languageCode }: NodeInputs,
  { logging, env, getBuildShipFile }: NodeScriptOptions,
): Promise<MenuItemOutput> {
  
  logging.log("Retrieving menu item in specific language:", { menuItemId, languageCode });

  if (!supabaseUrl || !supabaseKey) {
    return { 
      error: 'Supabase credentials required' 
    };
  }
  
  if (!menuItemId || typeof menuItemId !== 'number') {
    return { 
      error: 'Valid menuItemId required' 
    };
  }
  
  if (!languageCode || typeof languageCode !== 'string') {
    return { 
      error: 'Valid languageCode required' 
    };
  }

  const supabase = createClient(supabaseUrl, supabaseKey);

  try {
    const { data, error } = await supabase.rpc('get_menu_item_in_language', {
      p_menu_item_id: menuItemId,
      p_language_code: languageCode
    });

    if (error) {
      logging.log('Supabase error:', error.message);
      return {
        error: `Database error: ${error.message}`
      };
    }

    if (!data) {
      return {
        error: 'Menu item not found or no active SKU available'
      };
    }

    // =====================================================
    // NORMALIZE MODIFIER PRICING
    // =====================================================
    // Ensure consistent pricing structure across all modifiers
    const normalizedItem = {
      ...data,
      item_modifier_groups: (data.item_modifier_groups || []).map((group: any) => {
        return {
          ...group,
          modifiers: (group.modifiers || []).map((modifier: any) => {
            return {
              modifier_id: modifier.modifier_id,
              name: modifier.name,
              description: modifier.description,
              price: modifier.price ?? 0,
              display_order: modifier.display_order
            };
          })
        };
      })
    };

    logging.log('Menu item retrieved successfully:', {
      menuItemId: normalizedItem.menu_item_id,
      itemName: normalizedItem.item_name,
      languageCode: normalizedItem.language_code,
      modifierGroupCount: normalizedItem.item_modifier_groups?.length || 0
    });

    return normalizedItem;

  } catch (error: any) {
    logging.log('Processing error:', error.message);
    return {
      error: `Processing error: ${error.message}`
    };
  }
}
