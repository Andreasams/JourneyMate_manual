import fetch from 'node-fetch';
export default async function getSupabaseTableRow({
    supabaseUrl,
    supabaseKey,
    table,
    filter,
    token
}: NodeInputs, {auth}) : NodeOutput  {
    const apiKey = auth.getKey();
    if (apiKey === undefined) {
      throw new Error("Please select a key for this node.");
    }
    const url = `${supabaseUrl}/rest/v1/${table}?${encodeURI(filter)}`;

    const headers = {
        'apikey': apiKey,
        'Authorization': `Bearer ${token}`
    };

    const requestOptions = {
        method: 'GET',
        headers
    };

    const response = await fetch(url, requestOptions);
    const data = await response.json();

    return data;
}
