import { createClient } from '@supabase/supabase-js';

interface NodeInputs {
  supabaseUrl: string;
  supabaseKey: string;
  businessId: number;
  languageCode: string;
}

interface NodeScriptOptions {
  logging: any;
  env: any;
  getBuildShipFile: any;
}

interface NormalizedMenuOutput {
  menu_version: string;
  menu_items: any[];
  categories: any[];
  availableRestrictions: number[];
  availablePreferences: number[];
  availableAllergies: number[];
  availableCanBeMadeRestrictions: number[];
  availableCanBeMadePreferences: number[];
  error?: string;
}

/**
 * Normalized menu structure with explicit separation of dietary compliance types.
 * 
 * Processes menu data from get_menu_complete function and calculates which dietary 
 * types are available as "can be made" options.
 * 
 * Returns:
 * - availableRestrictions: Dietary types 1,3,4,5 available (IS or CAN BE MADE)
 * - availablePreferences: Dietary types 2,6,7 available (IS or CAN BE MADE)
 * - availableCanBeMadeRestrictions: Dietary types 1,3,4,5 available as CAN BE MADE
 * - availableCanBeMadePreferences: Dietary types 2,6,7 available as CAN BE MADE
 * 
 * This explicit separation enables client to make informed filtering decisions:
 * - Show items with dietary_type_ids match (inherently compliant)
 * - Show items with dietary_type_can_be_made_ids match (adaptable, ignore allergens)
 * 
 * Menu Item Authentic Languages:
 * Each menu item includes authentic_languages array from its parent Menu.
 * This array contains 1-3 language codes representing the original languages of the menu.
 * Enables "view in original language" feature by allowing client to query MenuItemTranslation
 * using the item's translation_id with any of the authentic language codes.
 */
export default async function getMenuComplete(
  { supabaseUrl, supabaseKey, businessId, languageCode }: NodeInputs,
  { logging, env, getBuildShipFile }: NodeScriptOptions,
): Promise<NormalizedMenuOutput> {
  
  logging.log("Processing normalized menu request with explicit dietary arrays:", { businessId, languageCode });

  if (!supabaseUrl || !supabaseKey) {
    return { 
      menu_version: '',
      menu_items: [], 
      categories: [], 
      availableRestrictions: [],
      availablePreferences: [], 
      availableAllergies: [],
      availableCanBeMadeRestrictions: [],
      availableCanBeMadePreferences: [],
      error: 'Supabase credentials required' 
    };
  }
  
  if (!businessId || typeof businessId !== 'number') {
    return { 
      menu_version: '',
      menu_items: [], 
      categories: [], 
      availableRestrictions: [],
      availablePreferences: [], 
      availableAllergies: [],
      availableCanBeMadeRestrictions: [],
      availableCanBeMadePreferences: [],
      error: 'Valid businessId required' 
    };
  }
  
  if (!languageCode) {
    return { 
      menu_version: '',
      menu_items: [], 
      categories: [], 
      availableRestrictions: [],
      availablePreferences: [], 
      availableAllergies: [],
      availableCanBeMadeRestrictions: [],
      availableCanBeMadePreferences: [],
      error: 'languageCode required' 
    };
  }

  const supabase = createClient(supabaseUrl, supabaseKey);

  try {
    const { data, error } = await supabase.rpc('get_menu_complete', {
      p_business_id: businessId,
      p_language_code: languageCode
    });

    if (error) {
      logging.log('Supabase error:', error.message);
      return {
        menu_version: '',
        menu_items: [],
        categories: [],
        availableRestrictions: [],
        availablePreferences: [],
        availableAllergies: [],
        availableCanBeMadeRestrictions: [],
        availableCanBeMadePreferences: [],
        error: `Database error: ${error.message}`
      };
    }

    if (!data) {
      return {
        menu_version: '',
        menu_items: [],
        categories: [],
        availableRestrictions: [],
        availablePreferences: [],
        availableAllergies: [],
        availableCanBeMadeRestrictions: [],
        availableCanBeMadePreferences: [],
        error: 'No menu data returned'
      };
    }

    // =====================================================
    // DIETARY TYPE CLASSIFICATION
    // =====================================================
    const RESTRICTION_IDS = new Set([1, 3, 4, 5]); // gluten-free, halal, lactose-free, kosher
    const PREFERENCE_IDS = new Set([2, 6, 7]);      // pescetarian, vegan, vegetarian

    // =====================================================
    // USE DATABASE FIELDS DIRECTLY
    // =====================================================
    const availableRestrictions = data.availableRestrictions || [];
    const availablePreferences = data.availablePreferences || [];

    // =====================================================
    // CALCULATE "CAN BE MADE" DIETARY TYPES FROM MENU ITEMS
    // =====================================================
    const allCanBeMadeIds = new Set<number>();
    
    (data.menu_items || []).forEach((item: any) => {
      const canBeMadeIds = item.dietary_type_can_be_made_ids || [];
      canBeMadeIds.forEach((id: number) => {
        allCanBeMadeIds.add(id);
      });
    });

    const availableCanBeMadeArray = Array.from(allCanBeMadeIds).sort((a, b) => a - b);

    const availableCanBeMadeRestrictions = availableCanBeMadeArray.filter(id => 
      RESTRICTION_IDS.has(id)
    );

    const availableCanBeMadePreferences = availableCanBeMadeArray.filter(id => 
      PREFERENCE_IDS.has(id)
    );

    // =====================================================
    // NORMALIZE MENU ITEMS: Pass through authentic_languages and normalize pricing
    // =====================================================
    const normalizedMenuItems = (data.menu_items || []).map((item: any) => {
      return {
        ...item,
        // authentic_languages is passed through directly from database
        // No transformation needed - already an array of language codes
        item_modifier_groups: (item.item_modifier_groups || []).map((group: any) => {
          return {
            ...group,
            modifiers: (group.modifiers || []).map((modifier: any) => {
              return {
                modifier_id: modifier.modifier_id,
                name: modifier.name,
                description: modifier.description,
                price: modifier.price ?? 0,
                display_order: modifier.display_order
              };
            })
          };
        })
      };
    });

    logging.log('Normalized menu processing completed:', { 
      version: data.menu_version || 'N/A',
      itemCount: normalizedMenuItems.length,
      categoryCount: data.categories?.length || 0,
      restrictionCount: availableRestrictions.length,
      preferenceCount: availablePreferences.length,
      canBeMadeRestrictionCount: availableCanBeMadeRestrictions.length,
      canBeMadePreferenceCount: availableCanBeMadePreferences.length,
      allergyCount: data.availableAllergies?.length || 0
    });

    return {
      menu_version: data.menu_version || '',
      menu_items: normalizedMenuItems,
      categories: data.categories || [],
      availableRestrictions,
      availablePreferences,
      availableAllergies: data.availableAllergies || [],
      availableCanBeMadeRestrictions,
      availableCanBeMadePreferences
    };

  } catch (error: any) {
    logging.log('Pass-through error:', error.message);
    return {
      menu_version: '',
      menu_items: [],
      categories: [],
      availableRestrictions: [],
      availablePreferences: [],
      availableAllergies: [],
      availableCanBeMadeRestrictions: [],
      availableCanBeMadePreferences: [],
      error: `Processing error: ${error.message}`
    };
  }
}
